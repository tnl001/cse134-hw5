function addFav(data, favRecipeArray) {
    favRecipeArray.push(data);
    localStorage.setItem("favRecipeArray", JSON.stringify(favRecipeArray));
}

This function adds a recipe to the Favorite list by first storing it inside an array and then storing the array inside localStorage.
